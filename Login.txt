--CUSTOMER INFO TABLE

  CREATE TABLE "PARAGON"."CUSTOMERS" 
   (	"CUSTOMER_ID" NUMBER(10,0) NOT NULL ENABLE, 
	"CUSTOMER_NAME" VARCHAR2(25 BYTE) NOT NULL ENABLE, 
	"CUSTOMER_CONTRACTS" VARCHAR2(255 BYTE) NOT NULL ENABLE, 
	"PARAGON_EMPLOYEE_ID" NUMBER(10,0) NOT NULL ENABLE, 
	"EMPLOYEE_START_DATE" DATE NOT NULL ENABLE, 
	"EMPLOYEE_END_DATE" DATE NOT NULL ENABLE, 
	"EMPLOYEE_BILL_RATE" NUMBER(6,2) NOT NULL ENABLE, 
	 CONSTRAINT "CUS_PK" PRIMARY KEY ("CUSTOMER_ID")

  TABLESPACE "USERS"  ENABLE,
